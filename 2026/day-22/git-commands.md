#####
git --version – Check installed Git version
git config --global user.name "Name" – Set username for commits
git config --global user.email "email" – Set email for commits
git config --global --list – View configured Git setting
mkdir devops-git-practice – Create new project folde
cd devops-git-practice – Move into project directory
git init – Initialize new Git repository
git status – Show current file status
ls -la – Show all files including hidden (.git)
touch git-commands.md – Create a new file
git add git-commands.md – Stage specific file
git add . – Stage all changes
git diff – Show unstaged changes
git commit -m "message" – Save staged changes
git log – Show full commit history
git log --oneline – Show compact commit history
git branch – Show available branches
git remote add origin <url> – Connect local repo to remote
git push -u origin main – Push commits to remote repository
